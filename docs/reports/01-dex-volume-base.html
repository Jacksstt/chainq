Reproducing this report

1. chainq_describe("dex.trades")
2. chainq_metric("dex_volume_usd", dimensions=["day"], filters={chain: "base"}, start="2026-01-01T00:00:00Z", end="2026-02-01T00:00:00Z")
3. chainq_metric("dex_protocol_share", filters={chain: "base"}, …)
4. chainq_chart_render(type="bar", data=…, x="day", y="volume_usd", filename="01-base-daily-volume.svg")
5. chainq_report(title=…, filename="01-dex-volume-base.html", sections=…)
